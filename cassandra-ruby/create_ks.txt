drop keyspace MovieLens;

create keyspace MovieLens with
  placement_strategy = 'org.apache.cassandra.locator.SimpleStrategy' AND
  strategy_options = {replication_factor:1};

use MovieLens;

create column family User with
  comparator = 'UTF8Type';

create column family User_MovieRatings with
  comparator = 'Int32Type';

create column family Movie with
  comparator = 'UTF8Type';

create column family MovieGenres with
  comparator = 'UTF8Type';

create column family MovieRatings with
  comparator='CompositeType(Int32Type,Int32Type,UTF8Type)';
